116 def show_statistics(self, statistics): # type: (Statistics) -> None 

117 """Format and print the statistics.""" 

118 for error_code in statistics.error_codes(): 

119 stats_for_error_code = statistics.statistics_for(error_code) 

120 statistic = next(stats_for_error_code) 

121 count = statistic.count 

122 count += sum(stat.count for stat in stats_for_error_code) 

123 self._write( 

124 "{count:<5} {error_code} {message}".format( 

125 count=count, 

126 error_code=error_code, 

127 message=statistic.message, 

128 ) 

129 )

131 def show_benchmarks(self, benchmarks): 

132 # type: (List[Tuple[str, float]]) -> None 

133 """Format and print the benchmarks.""" 

134 # NOTE(sigmavirus24): The format strings are a little confusing, even 

135 # to me, so here's a quick explanation: 

136 # We specify the named value first followed by a ':' to indicate we're 

137 # formatting the value. 

138 # Next we use '<' to indicate we want the value left aligned. 

139 # Then '10' is the width of the area. 

140 # For floats, finally, we only want only want at most 3 digits after 

141 # the decimal point to be displayed. This is the precision and it 

142 # can not be specified for integers which is why we need two separate 

143 # format strings. 

144 float_format = "{value:<10.3} {statistic}".format 

145 int_format = "{value:<10} {statistic}".format 

146 for statistic, value in benchmarks: 

147 if isinstance(value, int): 

148 benchmark = int_format(statistic=statistic, value=value) 

149 else: 

150 benchmark = float_format(statistic=statistic, value=value) 

151 self._write(benchmark)

153 def show_source(self, error): # type: (Violation) -> Optional[str] 

154 """Show the physical line generating the error. 

155 

156 This also adds an indicator for the particular part of the line that 

157 is reported as generating the problem. 

158 

159 :param error: 

160 This will be an instance of 

161 :class:`~flake8.style_guide.Violation`. 

162 :type error: 

163 flake8.style_guide.Violation 

164 :returns: 

165 The formatted error string if the user wants to show the source. 

166 If the user does not want to show the source, this will return 

167 ``None``. 

168 :rtype: 

169 str 

170 """ 

171 if not self.options.show_source or error.physical_line is None: 

172 return "" 

173 

174 # Because column numbers are 1-indexed, we need to remove one to get 

175 # the proper number of space characters. 

176 indent = "".join( 

177 c if c.isspace() else " " 

178 for c in error.physical_line[: error.column_number - 1] 

179 ) 

180 # Physical lines have a newline at the end, no need to add an extra 

181 # one 

182 return "{}{}^".format(error.physical_line, indent)

184 def _write(self, output): # type: (str) -> None

188 if self.output_fd is None or self.options.tee:
