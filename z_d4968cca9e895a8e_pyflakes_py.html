148 @classmethod 

149 def parse_options(cls, options): 

150 """Parse option values from Flake8's OptionManager.""" 

151 if options.builtins: 

152 cls.builtIns = cls.builtIns.union(options.builtins) 

153 cls.with_doctest = options.doctests 

154 

155 included_files = [] 

156 for included_file in options.include_in_doctest: 

157 if included_file == "": 

158 continue 

159 if not included_file.startswith((os.sep, "./", "~/")): 

160 included_files.append("./" + included_file) 

161 else: 

162 included_files.append(included_file) 

163 cls.include_in_doctest = utils.normalize_paths(included_files) 

164 

165 excluded_files = [] 

166 for excluded_file in options.exclude_from_doctest: 

167 if excluded_file == "": 

168 continue 

169 if not excluded_file.startswith((os.sep, "./", "~/")): 

170 excluded_files.append("./" + excluded_file) 

171 else: 

172 excluded_files.append(excluded_file) 

173 cls.exclude_from_doctest = utils.normalize_paths(excluded_files) 

174 

175 inc_exc = set(cls.include_in_doctest).intersection( 

176 cls.exclude_from_doctest
